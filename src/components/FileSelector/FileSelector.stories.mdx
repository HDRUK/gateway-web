import { Meta, Story, Canvas } from '@storybook/addon-docs';

import FileSelector from './FileSelector';
import LayoutBox from '../LayoutBox';
import Button from '../Button';

<Meta title='Organisms/FileSelector' component={FileSelector} />

<Canvas>
    <Story name='Multiple'>
        {props => {
            return (
                <LayoutBox width='600px'>
                    <FileSelector
                        actions={({ fileList }) => {
                            if (fileList.length) {
                                var reader = new FileReader();
                                reader.readAsText(fileList[0], 'UTF-8');
                                reader.onload = function (evt) {
                                    console.log(evt.target.result);
                                    try {
                                        console.log('JSON', JSON.parse(evt.target.result));
                                    } catch (e) {
                                        alert('there was an error: ' + e);
                                    }
                                };
                            }
                            return fileList.length ? <Button>Upload</Button> : null;
                        }}
                        noFilesMessage='No files have been selected'
                        multiple
                    />
                </LayoutBox>
            );
        }}
    </Story>
</Canvas>
