import { Meta, Story, Canvas } from '@storybook/addon-docs';
import Popover from './Popover';
import PopoverMenu from '../PopoverMenu';
import WastebinIcon from '../../images/icons/wastebin.svg';

<Canvas>

                    </Box>
                    <Box mb={6}>
                        <P weight='bold' mb={2}>
                            Custom trigger

                        <Popover {...props} trigger='My custom trigger' content={<div>My content</div>} />

                </>

                { label: 'Item 2', icon: () => <SVG src={WastebinIcon} />, action: () => console.log('item 2 clicked') },
